## Pipeline Architecture

### Component Data Flow

```
Isaac Sim (Sensors)
  ↓ ROS 2 Bridge
Stereo Images @ 60 FPS
  ↓ Isaac ROS
cuVSLAM Pose @ 60 FPS
  ↓ Nav2
Path Planning (&lt;200ms)
  ↓ Locomotion Controller
Motor Commands
```

### Latency Budget

| Component | Latency Target | Notes |
|-----------|---------------|--------|
| **Isaac Sim Rendering** | &lt;16ms | 60 FPS = 16.67ms/frame |
| **ROS 2 Bridge** | &lt;2ms | Topic pub/sub overhead |
| **Isaac ROS VSLAM** | &lt;16ms | GPU-accelerated processing |
| **Nav2 Local Planner** | &lt;50ms | DWB trajectory optimization |
| **Locomotion Controller** | &lt;16ms | Joint commands |
| **Total End-to-End** | **&lt;100ms** | Sensor capture → motor command |

## Step 1: Configure ROS 2 Bridge

Isaac Sim includes a built-in ROS 2 bridge for publishing sensor data.

### Enable ROS 2 Bridge in Isaac Sim

<CodeBlock language="python" title="isaac_sim_ros2_bridge.py">
{`#!/usr/bin/env python3
from omni.isaac.kit import SimulationApp
simulation_app = SimulationApp({"headless": False})

from omni.isaac.core import World
from omni.isaac.core.utils.stage import add_reference_to_stage
from omni.isaac.sensor import Camera
from omni.isaac.core.utils.extensions import enable_extension

# Enable ROS 2 bridge extension
enable_extension("omni.isaac.ros2_bridge")

from omni.isaac.ros2_bridge import ROS2Bridge

# Create world
world = World()
add_reference_to_stage("/Isaac/Environments/Simple_Warehouse/warehouse.usd", "/World/Warehouse")

# Add stereo cameras
camera_left = Camera(prim_path="/World/CameraLeft", position=[0, -0.06, 1.5], frequency=60, resolution=(640, 480))
camera_right = Camera(prim_path="/World/CameraRight", position=[0, 0.06, 1.5], frequency=60, resolution=(640, 480))
world.scene.add(camera_left)
world.scene.add(camera_right)

world.reset()

# Initialize ROS 2 bridge
ros2_bridge = ROS2Bridge()

# Publish camera topics
ros2_bridge.publish_camera_info(
    camera_prim_path="/World/CameraLeft",
    topic_name="/camera/left/camera_info",
    frame_id="camera_left"
)
ros2_bridge.publish_image(
    camera_prim_path="/World/CameraLeft",
    topic_name="/camera/left/image_raw",
    frame_id="camera_left"
)
ros2_bridge.publish_camera_info(
    camera_prim_path="/World/CameraRight",
    topic_name="/camera/right/camera_info",
    frame_id="camera_right"
)
ros2_bridge.publish_image(
    camera_prim_path="/World/CameraRight",
    topic_name="/camera/right/image_raw",
    frame_id="camera_right"
)

print("ROS 2 Bridge active. Publishing camera topics...")

while simulation_app.is_running():
    world.step(render=True)

simulation_app.close()`}
</CodeBlock>

## Step 2: Master Launch File

Integrate all components with a single launch file.

<CodeBlock language="python" title="full_navigation_system.launch.py">
{`from launch import LaunchDescription
from launch.actions import IncludeLaunchDescription, DeclareLaunchArgument
from launch.launch_description_sources import PythonLaunchDescriptionSource
from launch_ros.actions import Node
from ament_index_python.packages import get_package_share_directory
import os

def generate_launch_description():
    # Package directories
    isaac_ros_vslam_dir = get_package_share_directory('isaac_ros_visual_slam')
    nav2_bringup_dir = get_package_share_directory('nav2_bringup')
    biped_nav_dir = get_package_share_directory('biped_nav')

    return LaunchDescription([
        # 1. Isaac ROS VSLAM
        IncludeLaunchDescription(
            PythonLaunchDescriptionSource(
                os.path.join(isaac_ros_vslam_dir, 'launch', 'isaac_ros_visual_slam.launch.py')
            ),
            launch_arguments={
                'stereo_image_left_topic': '/camera/left/image_raw',
                'stereo_image_right_topic': '/camera/right/image_raw',
                'stereo_camera_info_left_topic': '/camera/left/camera_info',
                'stereo_camera_info_right_topic': '/camera/right/camera_info',
                'enable_slam_visualization': 'true'
            }.items()
        ),

        # 2. Nav2 Navigation Stack
        IncludeLaunchDescription(
            PythonLaunchDescriptionSource(
                os.path.join(biped_nav_dir, 'launch', 'biped_nav2.launch.py')
            )
        ),

        # 3. Health Monitor
        Node(
            package='health_monitor',
            executable='health_monitor_node',
            name='health_monitor',
            parameters=[{
                'monitor_topics': [
                    '/camera/left/image_raw',
                    '/visual_slam/tracking/odometry',
                    '/cmd_vel'
                ],
                'heartbeat_timeout': 1.0  # 1 second timeout
            }]
        ),

        # 4. RViz Visualization
        Node(
            package='rviz2',
            executable='rviz2',
            name='rviz2',
            arguments=['-d', os.path.join(biped_nav_dir, 'rviz', 'full_system.rviz')]
        )
    ])`}
</CodeBlock>

## Step 3: Health Monitoring

### Node Health Checker

<CodeBlock language="python" title="health_monitor.py">
{`#!/usr/bin/env python3
import rclpy
from rclpy.node import Node
from sensor_msgs.msg import Image
from nav_msgs.msg import Odometry
from geometry_msgs.msg import Twist
import time

class HealthMonitor(Node):
    def __init__(self):
        super().__init__('health_monitor')

        self.last_image_time = time.time()
        self.last_odom_time = time.time()
        self.last_cmd_time = time.time()

        # Subscriptions
        self.create_subscription(Image, '/camera/left/image_raw', self.image_callback, 10)
        self.create_subscription(Odometry, '/visual_slam/tracking/odometry', self.odom_callback, 10)
        self.create_subscription(Twist, '/cmd_vel', self.cmd_callback, 10)

        # Health check timer (1 Hz)
        self.create_timer(1.0, self.check_health)

    def image_callback(self, msg):
        self.last_image_time = time.time()

    def odom_callback(self, msg):
        self.last_odom_time = time.time()

    def cmd_callback(self, msg):
        self.last_cmd_time = time.time()

    def check_health(self):
        now = time.time()
        timeout = 1.0  # 1 second timeout

        camera_healthy = (now - self.last_image_time) < timeout
        vslam_healthy = (now - self.last_odom_time) < timeout
        nav_healthy = (now - self.last_cmd_time) < timeout

        if not camera_healthy:
            self.get_logger().error("❌ Camera feed lost!")
        if not vslam_healthy:
            self.get_logger().error("❌ VSLAM tracking lost!")
        if not nav_healthy:
            self.get_logger().warn("⚠️  Nav2 not publishing commands")

        if camera_healthy and vslam_healthy and nav_healthy:
            self.get_logger().info("✅ All systems healthy")

def main():
    rclpy.init()
    node = HealthMonitor()
    rclpy.spin(node)
    rclpy.shutdown()

if __name__ == '__main__':
    main()`}
</CodeBlock>

## Step 4: Performance Validation

### End-to-End Latency Measurement

<CodeBlock language="python" title="measure_latency.py">
{`#!/usr/bin/env python3
import rclpy
from rclpy.node import Node
from sensor_msgs.msg import Image
from geometry_msgs.msg import Twist
import time

class LatencyMeasure(Node):
    def __init__(self):
        super().__init__('latency_measure')
        self.create_subscription(Image, '/camera/left/image_raw', self.image_callback, 10)
        self.create_subscription(Twist, '/cmd_vel', self.cmd_callback, 10)

        self.last_image_timestamp = None
        self.latencies = []

    def image_callback(self, msg):
        # Record image timestamp
        self.last_image_timestamp = msg.header.stamp.sec + msg.header.stamp.nanosec * 1e-9

    def cmd_callback(self, msg):
        if self.last_image_timestamp is not None:
            cmd_time = time.time()
            latency = cmd_time - self.last_image_timestamp
            self.latencies.append(latency)

            if len(self.latencies) >= 100:
                avg_latency = sum(self.latencies) / len(self.latencies)
                max_latency = max(self.latencies)
                self.get_logger().info(f"Latency: Avg={avg_latency*1000:.1f}ms, Max={max_latency*1000:.1f}ms")
                self.latencies = []

def main():
    rclpy.init()
    node = LatencyMeasure()
    rclpy.spin(node)
    rclpy.shutdown()

if __name__ == '__main__':
    main()`}
</CodeBlock>

<Callout type="success" title="Target Latency">
**Acceptable Performance**:
- Average latency: &lt;80ms
- Maximum latency: &lt;100ms
- 99th percentile: &lt;120ms

If latencies exceed 150ms, troubleshoot:
1. Reduce VSLAM feature count
2. Lower Nav2 replanning frequency
3. Use headless Isaac Sim for lower render overhead
</Callout>

## Step 5: Autonomous Navigation Demo

### 10-Trial Success Rate Test

<CodeBlock language="python" title="navigation_test.py">
{`#!/usr/bin/env python3
import rclpy
from rclpy.node import Node
from geometry_msgs.msg import PoseStamped
from nav2_msgs.action import NavigateToPose
from rclpy.action import ActionClient
import numpy as np

class NavigationTester(Node):
    def __init__(self):
        super().__init__('navigation_tester')
        self.nav_client = ActionClient(self, NavigateToPose, 'navigate_to_pose')

    def run_test(self, num_trials=10):
        successes = 0
        goals = [
            [5.0, 0.0], [10.0, 5.0], [-5.0, 5.0],
            [0.0, 10.0], [8.0, -3.0], [-7.0, -4.0],
            [3.0, 7.0], [-3.0, -7.0], [12.0, 0.0], [0.0, 0.0]
        ]

        for i, goal_xy in enumerate(goals[:num_trials]):
            self.get_logger().info(f"Trial {i+1}/{num_trials}: Goal={goal_xy}")

            goal_msg = NavigateToPose.Goal()
            goal_msg.pose.header.frame_id = 'map'
            goal_msg.pose.pose.position.x = goal_xy[0]
            goal_msg.pose.pose.position.y = goal_xy[1]
            goal_msg.pose.pose.orientation.w = 1.0

            future = self.nav_client.send_goal_async(goal_msg)
            rclpy.spin_until_future_complete(self, future)

            goal_handle = future.result()
            if goal_handle.accepted:
                result_future = goal_handle.get_result_async()
                rclpy.spin_until_future_complete(self, result_future)
                result = result_future.result()

                if result.status == 4:  # SUCCEEDED
                    successes += 1
                    self.get_logger().info(f"✅ Trial {i+1} SUCCESS")
                else:
                    self.get_logger().warn(f"❌ Trial {i+1} FAILED")

        success_rate = (successes / num_trials) * 100
        self.get_logger().info(f"\\n🏆 Success Rate: {successes}/{num_trials} ({success_rate:.1f}%)")

        if success_rate >= 90:
            self.get_logger().info("✅ PASS: Success rate ≥90%")
        else:
            self.get_logger().warn("❌ FAIL: Success rate &lt;90%")

def main():
    rclpy.init()
    tester = NavigationTester()
    tester.run_test(num_trials=10)
    rclpy.shutdown()

if __name__ == '__main__':
    main()`}
</CodeBlock>
